Funding Trigger

  "raises Series A" OR "raises Series B" OR "funding round" OR "venture-backed" OR 
  "seed investment" OR "capital raise" OR "private equity investment" OR "IPO announced"
) [location]

  "expands to" OR "opens office in" OR "new branch" OR "launches operations" OR 
  "new market entry" OR "international expansion" OR "franchise opening" OR "regional hub"
) [location]

  "launches new platform" OR "introduces feature" OR "new release" OR "beta launch" OR 
  "rolls out update" OR "integration with" OR "product unveiled" OR "new app available"
) [location]
